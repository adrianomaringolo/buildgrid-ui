File Upload Dropzone

O FileUploadDropzone é um componente completo para upload de arquivos que oferece uma interface intuitiva com drag & drop, preview de imagens, controle de progresso e validação robusta. Ideal para formulários que precisam de upload de documentos, imagens ou qualquer tipo de arquivo.

Importação​

import { FileUploadDropzone } from 'buildgrid-ui'
import type { FileUploadConfig, UploadedFile } from 'buildgrid-ui'

Uso Básico​

import { FileUploadDropzone } from 'buildgrid-ui'
import { useState } from 'react'

function BasicFileUpload() {
  const [files, setFiles] = useState<UploadedFile[]>([])

  return (
    <FileUploadDropzone
      onFilesChange={setFiles}
      placeholder="Arraste arquivos aqui ou clique para selecionar"
    />
  )
}

Configuração Personalizada​

import { FileUploadDropzone } from 'buildgrid-ui'
import type { FileUploadConfig } from 'buildgrid-ui'

function CustomConfigUpload() {
  const config: FileUploadConfig = {
    maxFiles: 3,
    maxSize: 5 * 1024 * 1024, // 5MB
    acceptedFileTypes: ['image/*', 'application/pdf', '.doc', '.docx'],
    multiple: true,
    showPreview: true,
    showProgress: true
  }

  return (
    <FileUploadDropzone
      config={config}
      placeholder="Máximo 3 arquivos (imagens ou documentos)"
    />
  )
}
